#include "pch.h"
#include "Il2CppApi.h"
#include <cstdio>

IIl2CppRuntime* g_Runtime = nullptr;

static const char* TARGET_VERSION = "WLAAg5IQf2ffgegfxkpA";

typedef void (*fn_Ctor)(void* instance, void* version);
static IHook<fn_Ctor>* g_Hook = nullptr;

void hk_Ctor(void* instance, void* version)
{
    if (version && g_Runtime)
    {
        std::wstring orig = g_Runtime->ReadString(version);
        printf("[photon] .ctor: \"%ls\" -> \"%s\"\n", orig.c_str(), TARGET_VERSION);
        version = g_Runtime->NewString(TARGET_VERSION);
    }
    if (g_Hook && g_Hook->Original)
        g_Hook->Original(instance, version);
}

extern "C" __declspec(dllexport) void InitPlugin(IIl2CppRuntime* runtime)
{
    g_Runtime = runtime;
}

extern "C" __declspec(dllexport) void OnLoad()
{
    printf("[photon] OnLoad\n");

    void* image = g_Runtime->GetImage("AnimalCompany.dll");
    if (!image) { printf("[photon] failed to get image\n"); return; }

    void* cls = g_Runtime->GetClass(image, "AnimalCompany", "SetPhotonVersionAction");
    if (!cls) { printf("[photon] failed to get class\n");  return; }

    g_Hook = g_Runtime->HookMethod(cls, ".ctor", hk_Ctor, 1);
    printf("[photon] hook installed\n");
}
